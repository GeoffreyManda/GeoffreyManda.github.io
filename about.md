<script>
document.addEventListener('DOMContentLoaded', function() {
  const cvTrack = document.querySelector('.cv-carousel-track');
  const cvCards = Array.from(document.querySelectorAll('.cv-card'));
  const cvPrevBtn = document.querySelector('.cv-carousel-btn-prev');
  const cvNextBtn = document.querySelector('.cv-carousel-btn-next');
  const cvDotsContainer = document.querySelector('.cv-carousel-dots');

  if (!cvTrack || cvCards.length === 0) return;

  let cvCurrentIndex = 0;

  // Generate dots
  cvCards.forEach((_, index) => {
    const dot = document.createElement('button');
    dot.classList.add('cv-carousel-dot');
    dot.setAttribute('aria-label', `Go to milestone ${index + 1}`);
    if (index === 0) dot.classList.add('active');
    cvDotsContainer.appendChild(dot);
  });

  const cvDots = Array.from(cvDotsContainer.children);

  // Update card positions
  function updateCVCards() {
    cvCards.forEach((card, index) => {
      card.removeAttribute('data-position');

      const diff = index - cvCurrentIndex;

      if (diff === 0) {
        card.setAttribute('data-position', 'center');
      } else if (diff === -1 || (cvCurrentIndex === 0 && index === cvCards.length - 1)) {
        card.setAttribute('data-position', 'left');
      } else if (diff === 1 || (cvCurrentIndex === cvCards.length - 1 && index === 0)) {
        card.setAttribute('data-position', 'right');
      } else if (diff < -1) {
        card.setAttribute('data-position', 'hidden-left');
      } else {
        card.setAttribute('data-position', 'hidden-right');
      }
    });

    // Update dots
    cvDots.forEach((dot, index) => {
      dot.classList.toggle('active', index === cvCurrentIndex);
    });

    // Update dynamic title based on current card's data-title attribute
    const currentCard = cvCards[cvCurrentIndex];
    const titleText = currentCard.getAttribute('data-title') || 'Academic Journey';
    const titleElement = document.getElementById('dynamic-carousel-title');
    if (titleElement) {
      titleElement.textContent = titleText;
    }
  }

  // Next button
  cvNextBtn.addEventListener('click', () => {
    cvCurrentIndex = (cvCurrentIndex + 1) % cvCards.length;
    updateCVCards();
  });

  // Previous button
  cvPrevBtn.addEventListener('click', () => {
    cvCurrentIndex = (cvCurrentIndex - 1 + cvCards.length) % cvCards.length;
    updateCVCards();
  });

  // Dot navigation
  cvDots.forEach((dot, index) => {
    dot.addEventListener('click', () => {
      cvCurrentIndex = index;
      updateCVCards();
    });
  });

  // Auto-play carousel - advance every 10 seconds
  let autoPlayInterval = setInterval(() => {
    cvNextBtn.click();
  }, 10000);

  // Pause auto-play on hover
  const carouselContainer = document.querySelector('.cv-carousel-container');
  carouselContainer.addEventListener('mouseenter', () => {
    clearInterval(autoPlayInterval);
  });

  // Resume auto-play when mouse leaves
  carouselContainer.addEventListener('mouseleave', () => {
    autoPlayInterval = setInterval(() => {
      cvNextBtn.click();
    }, 10000);
  });

  // Pause auto-play when user interacts
  [cvPrevBtn, cvNextBtn, ...cvDots].forEach(element => {
    element.addEventListener('click', () => {
      clearInterval(autoPlayInterval);
      autoPlayInterval = setInterval(() => {
        cvNextBtn.click();
      }, 10000);
    });
  });

  // Initialize
  updateCVCards();
});
</script>
